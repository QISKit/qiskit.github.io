Instruction collection.

from ._instruction import Instruction
from ._qiskiterror import QISKitError



[ドキュメント]class InstructionSet(object):
    """Instruction collection."""


[ドキュメント]    def __init__(self):
        """New collection of instructions."""
        self.instructions = set([])



[ドキュメント]    def add(self, gate):
        """Add instruction to set."""
        if not isinstance(gate, Instruction):
            raise QISKitError("attempt to add non-Instruction" +
                                  " to InstructionSet")
        self.instructions.add(gate)



[ドキュメント]    def inverse(self):
        """Invert all instructions."""
        for instruction in self.instructions:
            instruction.inverse()
        return self



[ドキュメント]    def q_if(self, *qregs):
        """Add controls to all instructions."""
        for gate in self.instructions:
            gate.q_if(*qregs)
        return self



[ドキュメント]    def c_if(self, c, val):
        """Add classical control register to all instructions."""
        for gate in self.instructions:
            gate.c_if(c, val)
        return self
